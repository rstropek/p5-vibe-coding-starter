import "globals";

function setup() {
    createCanvas(500, 500);
}

function draw() {
    background("black");
    
    // Spaceship
    stroke("yellow");
    strokeWeight(2);
    fill("gold");
    triangle(100, 350, 100 - 20, 350 + 40, 100 + 20, 350 + 40);

    // Laser
    noFill();
    stroke("red");
    line(100, 300, 100, 300 - 20);

    // Meteor
    fill("orange");
    stroke("darkred");
    circle(300, 100, 30);
}
